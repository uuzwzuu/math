void fillbox(pair a) {

void drawbox(pair a, int x){
    draw(box(a, a+(1,1)));
    label(string(x), a+(0.5,0.5));
}

    for(int j=0; j<2*i+1; ++j){
          drawbox((j+3-i, 3-i), arr[counter]);
          ++ counter;
    }
}

    for(int j=0; j<2*i+1; ++j){
          drawbox((j+3-i, i-3), arr[counter]);
          ++ counter;
    }
}
